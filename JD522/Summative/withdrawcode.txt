    private void Withdraw() {
        try {
            // SQL Connection Stuff. Also grabs id from database and allows login.
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/atmdetails", "root", "TigerKiller_1");
            Statement stm = con.createStatement();
            String UserID;
            int IBal;
            String temp;
            Account obj = new Account();
            // Getts the ID from a temp file that was saved via the login pane
            File file = new File("temp.txt");
            BufferedReader br = new BufferedReader(new FileReader(file));
            temp = br.readLine();
            //End
            obj.setID(temp);
            UserID = obj.getID();
            IBal = Integer.parseInt(txtWAmount.getText());
            
          
            
            //Searching for the ID in database and updating the current value
            String sQuery = "UPDATE atmlog SET CURRENTBALANCE = CURRENTBALANCE - "+IBal+" WHERE ID= "+UserID+"";
            stm.execute(sQuery);
            
            JOptionPane.showMessageDialog(this, "You Withdraw R" + IBal );
            
            con.close();
        }catch (Exception e) {
            JOptionPane.showMessageDialog(this, e);  
        }
        
    }
