    private void CheckBalance() {
        try {
            // SQL Connection Stuff. Also grabs id from database and allows login.
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/atmdetails", "root", "TigerKiller_1");
            PreparedStatement st = null;
            ResultSet rs = null;
            String UserID;
            String temp;
            Account obj = new Account();
            // Getts the ID from a temp file that was saved via the login pane
            File file = new File("temp.txt");
            BufferedReader br = new BufferedReader(new FileReader(file));
            temp = br.readLine();
            //End
            obj.setID(temp);
            UserID = obj.getID();
            
            
            //Searching for the ID in database
            String sQuery = "SELECT * FROM atmlog where ID= "+UserID+"";
            st = con.prepareStatement(sQuery);
            rs = st.executeQuery();
            
            //Checking if ID is valid
             while (rs.next()) {
                int id = rs.getInt("ID");
                int Balance = rs.getInt("CURRENTBALANCE");
                
                if (Integer.parseInt(UserID) == id) {
                    JOptionPane.showMessageDialog(this, "Your Current Balance is R" + Balance );
                    con.close();
                    break;
                }
             }
            
        }catch (Exception e) {
            JOptionPane.showMessageDialog(this, e);  
        }
        
    }
    
    
    private void Statement() {
        try {
            // SQL Connection Stuff. Also grabs id from database and allows login.
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/atmdetails", "root", "TigerKiller_1");
            PreparedStatement st = null;
            ResultSet rs = null;
            String UserID;
            String temp;
            Account obj = new Account();
            // Getts the ID from a temp file that was saved via the login pane
            File file = new File("temp.txt");
            BufferedReader br = new BufferedReader(new FileReader(file));
            temp = br.readLine();
            //End
            obj.setID(temp);
            UserID = obj.getID();
            
            
            //Searching for the ID in database
            String sQuery = "SELECT * FROM atmlog where ID= "+UserID+"";
            st = con.prepareStatement(sQuery);
            rs = st.executeQuery();
            
            //Checking if ID is valid
             while (rs.next()) {
                int id = rs.getInt("ID");
                String Username = rs.getString("USERNAME");
                int Balance = rs.getInt("CURRENTBALANCE");
                int InitBal = rs.getInt("INITIALBALANCE");
                
                if (Integer.parseInt(UserID) == id) {
                    
                    JOptionPane.showMessageDialog(this,
                            "\n ID: " + id + 
                            "\n Name: " + Username + 
                            "\n Initial Balance R" + InitBal + 
                            "\n Current Balance R" + Balance
                    );
                    con.close();
                    break;
                }
             }
            
        }catch (Exception e) {
            JOptionPane.showMessageDialog(this, e);  
        }
        
    }
